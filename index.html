Code Walkthrough

Python / Pandas 
import pandas as pd
import numpy as np

# Load the auction dataset
df = pd.read_csv("data/ipl_2026_auction.csv")

# Quick shape & column inspection
print(df.shape)       # (244, 18)
print(df.dtypes)

# Null analysis
null_report = pd.DataFrame({
    "null_count": df.isnull().sum(),
    "null_%": (df.isnull().mean() * 100).round(2)
})
print(null_report[null_report["null_count"] > 0])
Python / Pandas — Cleaning & Feature Engineering 
# Fill nulls for unsold players (Sold_Price_Cr = NaN → 0)
df["Sold_Price_Cr"] = df["Sold_Price_Cr"].fillna(0)

# Clip statistical outliers to realistic T20 ranges
df["T20_Strike_Rate"] = df["T20_Strike_Rate"].clip(70, 220)
df["T20_Economy"]     = df["T20_Economy"].clip(4, 16)

# Feature Engineering
df["Price_Multiple"] = (
    df["Sold_Price_Cr"] / df["Base_Price_Cr"]
).round(2)

df["Experience_Band"] = pd.cut(
    df["IPL_Experience_Years"],
    bins=[-1, 0, 2, 5, 10, 25],
    labels=["Debutant", "Emerging", "Established", "Veteran", "Legend"]
)
Python / Pandas — EDA 
sold_df = df[df["Sold"] == True].copy()

# Sold rate per role
role_sold = df.groupby("Role")["Sold"].agg(
    sold="sum", total="count", rate="mean"
)

# Team-wise spending breakdown
team_spend = (
    sold_df.groupby("Winning_Team")
    .agg(
        Players=("Player_Name", "count"),
        Total_Cr=("Sold_Price_Cr", "sum"),
        Max_Bid=("Sold_Price_Cr", "max"),
    )
    .sort_values("Total_Cr", ascending=False)
)

# Correlation with price
corr = sold_df[[
    "Sold_Price_Cr", "T20_Batting_Avg", "T20_Strike_Rate",
    "T20_Wickets", "IPL_Experience_Years"
]].corr()["Sold_Price_Cr"]
Python / Pandas — Finding Hidden Gems 
# Composite gem score: high performance, low price
sold_df["Gem_Score"] = (
    (sold_df["T20_Batting_Avg"]  / sold_df["T20_Batting_Avg"].max()) * 0.3
  + (sold_df["T20_Strike_Rate"] / sold_df["T20_Strike_Rate"].max()) * 0.3
  + (sold_df["T20_Wickets"]     / sold_df["T20_Wickets"].max())     * 0.2
  - (sold_df["Sold_Price_Cr"]   / sold_df["Sold_Price_Cr"].max())   * 0.2
)

hidden_gems = sold_df.nlargest(10, "Gem_Score")[[
    "Player_Name", "Role", "T20_Batting_Avg",
    "T20_Strike_Rate", "Sold_Price_Cr", "Winning_Team"
]]
